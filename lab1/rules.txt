# База правил для системы подбора предметов League of Legends
# Формат: ЕСЛИ условие1 [И/ИЛИ условие2] ТО объект=значение

# Правила для анализа команды врага
ЕСЛИ количество_врагов_ап >= 3 ТО рекомендация=предметы_против_ап
ЕСЛИ количество_врагов_танк >= 3 ТО рекомендация=предметы_против_танк
ЕСЛИ количество_врагов_крит >= 2 И тип_персонажа = танк ТО рекомендация=предметы_против_крит
ЕСЛИ количество_врагов_контроль >= 3 И тип_урона = магический ТО рекомендация=магические_предметы_против_контроль
ЕСЛИ количество_врагов_контроль >= 3 И тип_урона = физический ТО рекомендация=физические_предметы_против_контроль
ЕСЛИ количество_врагов_хил >= 2 И тип_урона = магический ТО рекомендация=магические_предметы_против_хил
ЕСЛИ количество_врагов_хил >= 2 И тип_урона = физический ТО рекомендация=физические_предметы_против_хил
ЕСЛИ количество_врагов_щит >= 3 ТО рекомендация=предметы_против_щит
ЕСЛИ количество_врагов_мало_хп >= 3 И тип_урона = физический ТО рекомендация=леталити
ЕСЛИ количество_врагов_мало_хп >= 3 И тип_урона = магический ТО рекомендация=бурст

# Правила для выбора предметов по роли персонажа
ЕСЛИ тип_персонажа = энч ТО рекомендация=предметы_для_энч
ЕСЛИ тип_персонажа = крит ТО рекомендация=предметы_для_крит
ЕСЛИ тип_персонажа = танк ТО рекомендация=предметы_для_танк
ЕСЛИ тип_персонажа = мало_хп ТО рекомендация=предметы_от_бурст
ЕСЛИ тип_персонажа = мало_хп ТО рекомендация=предметы_от_леталити
