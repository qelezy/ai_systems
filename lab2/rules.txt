# База правил для системы подбора предметов League of Legends

ЕСЛИ количество_врагов_ап >= 3 ТО тип_угрозы=магическая
ЕСЛИ количество_врагов_танк >= 3 ТО тип_угрозы=танковая
ЕСЛИ количество_врагов_крит >= 2 ТО тип_угрозы=крит_урон
ЕСЛИ количество_врагов_контроль >= 3 ТО тип_угрозы=контроль
ЕСЛИ количество_врагов_хил >= 2 ТО тип_угрозы=хил
ЕСЛИ количество_врагов_щит >= 3 ТО тип_угрозы=щит
ЕСЛИ количество_врагов_мало_хп >= 3 ТО тип_угрозы=легкие_цели

ЕСЛИ тип_угрозы=магическая ТО рекомендация=предметы_против_ап
ЕСЛИ тип_угрозы=танковая ТО рекомендация=предметы_против_танк
ЕСЛИ тип_угрозы=крит_урон И тип_персонажа = танк ТО рекомендация=предметы_против_крит
ЕСЛИ тип_угрозы=контроль И тип_урона = ап ТО рекомендация=магические_предметы_против_контроль
ЕСЛИ тип_угрозы=контроль И тип_урона = ад ТО рекомендация=физические_предметы_против_контроль
ЕСЛИ тип_угрозы=хил И тип_урона = ап ТО рекомендация=магические_предметы_против_хил
ЕСЛИ тип_угрозы=хил И тип_урона = ад ТО рекомендация=физические_предметы_против_хил
ЕСЛИ тип_угрозы=щит ТО рекомендация=предметы_против_щит
ЕСЛИ тип_угрозы=легкие_цели И тип_урона = ад ТО рекомендация=предметы_леталити
ЕСЛИ тип_угрозы=легкие_цели И тип_урона = ап ТО рекомендация=предметы_бурст

ЕСЛИ тип_персонажа = энч ТО рекомендация=предметы_для_энч
ЕСЛИ тип_персонажа = крит ТО рекомендация=предметы_для_крит
ЕСЛИ тип_персонажа = танк ТО рекомендация=предметы_для_танк
ЕСЛИ тип_персонажа = мало_хп ТО рекомендация=предметы_от_бурст
ЕСЛИ тип_персонажа = мало_хп ТО рекомендация=предметы_от_леталити
